	\section{An Algorithm Outputting A Regular Expression In The Rev Language}
	In the below pseudocode the language defined by $r$ is over the alphabet $\Sigma$.
	
	\begin{algorithm}[!h]
		\caption{Reverses a regular expression $r$.}
		\begin{algorithmic}[1]
		\Procedure {Reverse}{$r$}
			\If {$r = \epsilon$}
				\State \Return r
			\ElsIf {$r = \emptyset$}
				\State \Return r
			\ElsIf {$r \in \Sigma$}
				\State \Return r
			\ElsIf {$r = r_1 r_2$}
				\State \Return \textsc{Reverse}($r_2$)\textsc{Reverse}($r_1$)
			\ElsIf {$r = r_1|r_2$}
				\State \Return \textsc{Reverse}($r_1$)$|$\textsc{Reverse}($r_2$)
			\ElsIf {$r = r_1\ast$}
				\State \Return \textsc{Reverse}($r_1$)$\ast$
			\ElsIf {$r = (r_1)$}
				\State \Return (\textsc{Reverse}($r_1$))
			\EndIf
		\EndProcedure
		\end{algorithmic}
	\end{algorithm}
	
	\section{Regular Expressions For Lexical Analysers}
	\begin{itemize}
		\item Identifiers that must start with a letter or an underscore and can continue with letters, underscores, or digits: {\verb [a-zA-Z_][a-zA-Z_0-9]* }
		\item Integers that contain one or more digits: {\verb [0-9]+ }
		\item Real numbers that contain zero or more digits followed by a decimal point followed by one or more digits: {\verb [0-9]*\.[0-9]+ }
		\item Comments that are delimited by curly braces; comments cannot be nested: {\verb \{.+\} }
	\end{itemize}
